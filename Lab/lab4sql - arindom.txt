N04:


SELECT 	ITEM.ITEM_ID, ITEM.ITEM_NAME, SUM(RELATION.QTY_SOLD) AS SOLD_ITEM, ITEM.ITEM_PRICE ,SUM(RELATION.QTY_SOLD*ITEM.ITEM_PRICE) AS FINAL_AMOUNT 

FROM SALE, ITEM, RELATION 

WHERE SALE.BILL_NO = RELATION.BILL_NO AND ITEM.ITEM_ID = RELATION.ITEM_ID 

GROUP BY (RELATION.ITEM_ID)
